# SPA 三页合一 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 index.html、knowledge.html、tools.html 合并为单页 SPA，GSAP 驱动 Apple 风格缩放位移过渡。

**Architecture:** 共享 nav-bar + `.view-container` 包含三个 `#view-*` 面板。ViewManager 管理懒初始化和 GSAP timeline 过渡。CSS 命名空间隔离各视图布局。

**Tech Stack:** 纯原生 HTML/CSS/JS，GSAP 3.x + ScrollTrigger，项目已有 lib（miaosite-auth.js, weather-state.js, raindrop-*.js）。

## Global Constraints

- 纯原生 JS，禁止 React/Vue 框架
- 单一 URL 策略，不改变地址栏
- 懒初始化：首屏只加载 home 视图 JS
- prefers-reduced-motion 时跳过动画
- 保留所有现有功能（条码/OCR、Wiki CRUD、AI Chat）

### Task 4: 实现 ViewManager 路由 + 共享 JS

**Files:**
- Modify: `index.html`（在共享 `<script>` 块中实现 ViewManager）

**Consumes:** Task 2 的 nav-bar（`data-view` buttons）、Task 3 的 `initHomeView()`
**Produces:** `ViewManager` 单例、nav-pill 指示器动画、共享天气/在线人数 SSE、`prefers-reduced-motion` 处理

- [ ] **Step 1: 实现 ViewManager 核心**

在 `index.html` 的共享 `<script>` 块中写入（在库导入之后、视图初始化之前）：

```js
// ============================================================
// ViewManager — 视图路由 + GSAP 过渡
// ============================================================
(function() {
  var views = {};
  var initializers = {};
  var initialized = {};
  var current = 'home';
  var isTransitioning = false;
  var prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  // 监听 reduced-motion 变化
  window.matchMedia('(prefers-reduced-motion: reduce)').addEventListener('change', function(e) {
    prefersReduced = e.matches;
  });

  function getTransitionOrigin() {
    var activeTab = document.querySelector('.nav-pill-tab.active');
    if (!activeTab) return { x: window.innerWidth / 2, y: 60 };
    var rect = activeTab.getBoundingClientRect();
    return { x: rect.left + rect.width / 2, y: rect.top + rect.height / 2 };
  }

  function animateView(fromName, toName) {
    var fromView = views[fromName];
    var toView = views[toName];
    if (!fromView || !toView) return Promise.resolve();

    if (prefersReduced) {
      fromView.classList.remove('active', 'leaving');
      toView.classList.add('active');
      document.body.setAttribute('data-active-view', toName);
      ScrollTrigger.refresh();
      return Promise.resolve();
    }

    var origin = getTransitionOrigin();
    gsap.set([fromView, toView], { transformOrigin: origin.x + 'px ' + origin.y + 'px' });

    fromView.classList.add('leaving');
    toView.classList.add('entering');

    return new Promise(function(resolve) {
      var tl = gsap.timeline({
        onComplete: function() {
          gsap.set([fromView, toView], { clearProps: 'transform,opacity' });
          fromView.classList.remove('active', 'leaving');
          toView.classList.add('active');
          toView.classList.remove('entering');
          document.body.setAttribute('data-active-view', toName);
          ScrollTrigger.refresh();
          resolve();
        }
      });

      tl.to(fromView, { scale: 0.92, opacity: 0, duration: 0.25, ease: 'power2.in' }, 0);
      tl.fromTo(toView, { scale: 0.95, opacity: 0 }, { scale: 1, opacity: 1, duration: 0.3, ease: 'expo.out' }, 0.15);

      // 背景平滑过渡
      var bgEls = document.querySelectorAll('.bg-yellow-glow, .tools-bg-glow, .tools-bg-noise');
      tl.to(bgEls, { opacity: 0, duration: 0.15 }, 0);
      tl.to(bgEls, { opacity: 1, duration: 0.2 }, 0.18);
    });
  }

  function updateNavTitle(name) {
    var titles = {
      home: { title: '喵码生成器', subtitle: '🏠 miaojiaozhu.com　💬 京me群 10229406786' },
      knowledge: { title: '蜜柑苏打', subtitle: '· 知识库' },
      tools: { title: '喵码 · AI 工具箱', subtitle: '小米Mimo v2.5 驱动' },
    };
    var info = titles[name];
    if (!info) return;
    var titleEl = document.getElementById('nav-title');
    var subtitleEl = document.getElementById('nav-subtitle');
    if (titleEl) titleEl.textContent = info.title;
    if (subtitleEl) subtitleEl.textContent = info.subtitle;
  }

  function updateNavPill(name) {
    document.querySelectorAll('.nav-pill-tab').forEach(function(tab) {
      var isActive = tab.getAttribute('data-view') === name;
      tab.classList.toggle('active', isActive);
      tab.setAttribute('aria-selected', isActive ? 'true' : 'false');
    });

    // 移动 GSAP 指示器
    var activeTab = document.querySelector('.nav-pill-tab.active');
    var indicator = document.getElementById('nav-pill-indicator');
    var lamp = document.getElementById('nav-pill-lamp');
    if (activeTab && indicator && lamp) {
      var rect = activeTab.getBoundingClientRect();
      var parentRect = activeTab.parentElement.getBoundingClientRect();
      var left = rect.left - parentRect.left;
      var width = rect.width;
      gsap.to(indicator, { x: left, width: width, duration: 0.35, ease: 'expo.out' });
      gsap.to(lamp, { x: left, duration: 0.35, ease: 'expo.out' });
    }
  }

  window.ViewManager = {
    current: current,

    register: function(name, initFn) {
      initializers[name] = initFn;
      views[name] = document.getElementById('view-' + name);
    },

    switchTo: function(name) {
      if (isTransitioning) return;
      if (name === current) return;

      // 懒初始化
      if (!initialized[name] && initializers[name]) {
        initializers[name]();
        initialized[name] = true;
      }

      isTransitioning = true;
      updateNavPill(name);
      updateNavTitle(name);

      var fromName = current;
      current = name;

      animateView(fromName, name).then(function() {
        isTransitioning = false;
      });
    },

    getCurrent: function() { return current; },
    isInitialized: function(name) { return !!initialized[name]; },
  };
})();
```

```js
  var countEl = document.getElementById('online-count');
  var badgeEl = document.getElementById('online-badge');
  if (!countEl) return;

  function updateCount(n) {
    countEl.textContent = n;
    if (badgeEl) {
      badgeEl.style.display = (n >= 0) ? '' : 'none';
    }
  }

  function setupPolling() {
    function poll() {
      fetch('/api/online-count')
        .then(function(r) { return r.ok ? r.json() : null; })
        .then(function(d) { if (d && typeof d.count === 'number') updateCount(d.count); })
        .catch(function() {})
        .finally(function() { setTimeout(poll, 30000); });
    }
    poll();
  }

  // SSE 优先
  try {
    var es = new EventSource('/api/online-count-sse');
    es.onmessage = function(e) {
      try { var d = JSON.parse(e.data); if (typeof d.count === 'number') updateCount(d.count); } catch(_) {}
    };
    es.onerror = function() { es.close(); setupPolling(); };
  } catch(_) {
    setupPolling();
  }
})();
```

```js
          window.WeatherState.setRainState(true);
          window._rainOverlay = new window.RaindropOverlay();
          window._rainOverlay.start();
        } else {
          window.WeatherState.setRainState(false);
        }
      })
      .catch(function() {});
```
